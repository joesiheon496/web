```html
<!DOCTYPE html>
<html lang="ko">
<head>
  <meta charset="utf-8" />
  <title>relative</title>
  <link rel="stylesheet" href="stylesheet.css">
</head>

  <div class="red box">RED</div>
  <div class="blue box">BLUE</div>
```
